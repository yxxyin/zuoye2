void my_initproctitle(char* argv[], char** last)
{
        int i = 0;
        char* p_tmp = NULL;
        size_t i_size = 0;
 
        for(i = 0; environ[i]; i++){
                i_size += strlen(environ[i]) + 1;
        }
 
        p_tmp = (char *)malloc(i_size);
        if(p_tmp == NULL){
                return ;
        }
 
        *last = argv[0];
        for(i = 0; argv[i]; i++){
                *last += strlen(argv[i]) + 1;
        }
 
        for(i = 0; environ[i]; i++){
                i_size = strlen(environ[i]) + 1;
                *last += i_size;
 
                strncpy(p_tmp, environ[i], i_size);
                environ[i] = p_tmp;
                p_tmp += i_size;
        }
 
        (*last)--;
 
        return ;
}
